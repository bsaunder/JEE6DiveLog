25  import java.util.Date;
26  
27  import net.bryansaunders.jee6divelog.model.DiveLogEntity;
28  import net.bryansaunders.jee6divelog.model.UserAccount;
29  import net.bryansaunders.jee6divelog.security.Identity;
30  
31  /**
32   * User Account Utilities.
33   * 
34   * @author Bryan Saunders <btsaunde@gmail.com>
35   * 
36   */
37  public final class AccountUtils {
38  
39      /**
40       * Default Private Constructor.
41       */
42      private AccountUtils() {
43          // Do Nothing
44      }
45  
46      /**
47       * Generates a UserAccount Object for the Current Identity.
48       * 
49       * @param identity
50       *            Identity to Copy
51       * @return UserAccount
52       */
53      public static UserAccount createUserAccount(final Identity identity) {
54          final UserAccount userAccount = new UserAccount();
55          userAccount.setEmail(identity.getUsername());
56          userAccount.setPublicApiKey(identity.getPublicApiKey());
57          userAccount.setPrivateApiKey(identity.getPrivateApiKey());
58          userAccount.setApiKeyExpiration(identity.getApiKeyExpiration());
59          userAccount.setRoles(identity.getRoles());
60          userAccount.setPermissions(identity.getPermissions());
61          return userAccount;
62      }
63  
64      /**
65       * Generates a Cleaned UserAccount Object for the Current Identity.
66       * 
67       * @param identity
68       *            Identity to Copy
69       * @return UserAccount
70       */
71      public static UserAccount createCleanUserAccount(final Identity identity) {
72          UserAccount account = AccountUtils.createUserAccount(identity);
73          account = AccountUtils.getCleanUserAccount(account);
74  
75          return account;
76      }
77      
78      /**
79       * Clears the Dates on any DiveLogEntity.
80       * 
81       * @param <T>
82       *            Generic Type
83       * @param entity
84       *            Entity to Clear Dates on
85       * @return Clean DiveLogEntity
86       */
87      public static <T extends DiveLogEntity> T getCleanEntity(final T entity) {
88          entity.setCreated(new Date());
89          entity.setUpdated(new Date());
90          return entity;
91      }
92  
93      /**
94       * Clears the Dates & Password on any UserAccount.
95       * 
96       * @param userAccount
97       *            UserAccount to Clear
98       * @return Clean UserAccount
99       */
100     public static UserAccount getCleanUserAccount(final UserAccount userAccount) {
101         userAccount.setPassword("***");
102         return AccountUtils.getCleanEntity(userAccount);
103     }
104 }
