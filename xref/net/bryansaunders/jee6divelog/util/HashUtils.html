33  import javax.crypto.spec.SecretKeySpec;

38  import org.slf4j.LoggerFactory;

40  /**
41   * Utility Methods for Dealing with Hashing.
42   * 
43   * @author Bryan Saunders <btsaunde@gmail.com>
44   * 
45   */
46  public final class HashUtils {
47      
48      /**
49       * Logger.
50       */
51      private static final Logger LOGGER = LoggerFactory.getLogger(HashUtils.class);
52  
53      /**
54       * Default Private Constructor.
55       */
56      private HashUtils() {
57          // Do Nothing
58      }
59  
60      /**
61       * Hashes the string using SHA-256 Algorithim.
62       * 
63       * @param stringToHash
64       *            String to hash
65       * @return Hashed String
66       */
67      public static String toSha256(final String stringToHash) {
68          return DigestUtils.sha256Hex(stringToHash);
69      }
70  
71      /**
72       * Encodes a string as Base64.
73       * 
74       * @param stringToEncode
75       *            String to encode
76       * @return Encoded String
77       */
78      public static String base64Encode(final String stringToEncode) {
79          return Base64.encodeBase64String(stringToEncode.getBytes());
80      }
81  
82      /**
83       * Decodes a Base64 string.
84       * 
85       * @param stringToDecode
86       *            String to decode
87       * @return Decoded String
88       */
89      public static String base64Decode(final String stringToDecode) {
90          return new String(Base64.decodeBase64(stringToDecode));
91      }
92  
93      /**
94       * Generates a HMAC-SHA1 Hash of the given String using the specified key.
95       * 
96       * @param stringToHash
97       *            String to Hash
98       * @param key
99       *            Key to Sign the String with
100      * @return Hashed String
101      */
102     public static String toHmacSha1(final String stringToHash, final String key) {
103         String result = "";
104         try {
105             // Get an hmac_sha1 key from the raw key bytes
106             final byte[] keyBytes = key.getBytes();
107             final SecretKeySpec signingKey = new SecretKeySpec(keyBytes, "HmacSHA1");
108 
109             // Get an hmac_sha1 Mac instance and initialize with the signing key
110             final Mac mac = Mac.getInstance("HmacSHA1");
111             mac.init(signingKey);
112 
113             // Compute the hmac on input data bytes
114             final byte[] rawHmac = mac.doFinal(stringToHash.getBytes());
115 
116             // Convert raw bytes to Hex
117             final byte[] base64 = Base64.encodeBase64(rawHmac);
118 
119             // Covert array of Hex bytes to a String
120             result = new String(base64);
121         } catch (GeneralSecurityException e) {
122             HashUtils.LOGGER.error("An Error Occured Generating an HMAC-SHA1 Hash!", e);
123         }
124 
125         return result;
126     }
