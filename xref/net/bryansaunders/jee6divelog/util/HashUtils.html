30  import org.apache.commons.codec.binary.Base64;
31  import org.apache.commons.codec.digest.DigestUtils;
32  
33  /**
34   * Utility Methods for Dealing with Hashing.
35   * 
36   * @author Bryan Saunders <btsaunde@gmail.com>
37   * 
38   */
39  public final class HashUtils {
40  
41      /**
42       * Default Private Constructor.
43       */
44      private HashUtils() {
45          // Do Nothing
46      }
47  
48      /**
49       * Hashes the string using SHA-256 Algorithim.
50       * 
51       * @param stringToHash
52       *            String to hash
53       * @return Hashed String
54       */
55      public static String toSha256(final String stringToHash) {
56          return DigestUtils.sha256Hex(stringToHash);
57      }
58  
59      /**
60       * Encodes a string as Base64.
61       * 
62       * @param stringToEncode
63       *            String to encode
64       * @return Encoded String
65       */
66      public static String base64Encode(final String stringToEncode) {
67          return Base64.encodeBase64String(stringToEncode.getBytes());
68      }
69  
70      /**
71       * Decodes a Base64 string.
72       * 
73       * @param stringToDecode
74       *            String to decode
75       * @return Decoded String
76       */
77      public static String base64Decode(final String stringToDecode) {
78          return new String(Base64.decodeBase64(stringToDecode));
79      }
80  }
