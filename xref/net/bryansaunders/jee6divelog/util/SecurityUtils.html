81      /**
82       * Determines if the Method is Secure or Not based on its Declared Annotations.
83       * 
84       * @param resourceMethod
85       *            Called Method
86       * @return true if the Method is Secured
87       */
88      public static boolean isMethodSecure(final ResourceMethod resourceMethod) {
89          boolean isSecured = false;
90          final Method method = resourceMethod.getMethod();
91          final Annotation[] declaredAnnotations = method.getDeclaredAnnotations();
92          for (final Annotation annotation : declaredAnnotations) {
93              if (SecurityUtils.isAnnotationSecure(annotation)) {
94                  isSecured = true;
95                  break;
96              }
97  
98          }
99          return isSecured;
100     }

102     /**
103      * Determines is the given Annotation is Secured.
104      * 
105      * @param annotation
106      *            Annotation to check
107      * @return true if the annotation is secured, false if it is not
108      */
109     public static boolean isAnnotationSecure(final Annotation annotation) {
110         boolean isSecured = false;
111         if (annotation instanceof HasPermission) {
112             isSecured = true;
113         } else if (annotation instanceof HasPermissions) {
114             isSecured = true;
115         } else if (annotation instanceof HasRole) {
116             isSecured = true;
117         } else if (annotation instanceof HasRoles) {
118             isSecured = true;
119         }
120         return isSecured;
121     }
