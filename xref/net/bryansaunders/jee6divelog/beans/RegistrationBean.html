116      * @return Registration Status
117      */
118     public String submitRegistration() {
119         String registrationResult = RegistrationBean.SUCCESS;
120 
121         final UserAccount user = this.createUser();
122         final UserAccount savedUser = this.userService.createUser(user);
123 
124         if (savedUser == null) {
125             registrationResult = RegistrationBean.FAILURE;
126         }
127 
128         return registrationResult;
129     }
