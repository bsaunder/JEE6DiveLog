56      /**
57       * Creates a new User.
58       * 
59       * @param user
60       *            User to Create.
61       * @return Saved user object, or null if an error occured.
62       */
63      public UserAccount createUser(final UserAccount user) {
64          UserAccount savedUser = null;
65  
66          try {
67              // Clear out Base Fields to ensure we save a new entity.
68              user.setId(null);
69              user.setVersion(null);
70              user.setCreated(null);
71              user.setUpdated(null);
72  
73              // Set Default Roles
74              final List<Role> defaultRoles = new LinkedList<Role>();
75              defaultRoles.add(Role.USER);
76              user.setRoles(defaultRoles);
77  
78              // Set Default Permissions
79              final List<Permission> defaultPermissions = Permission.getDefaults(Role.USER);
80              user.setPermissions(defaultPermissions);
81  
82              savedUser = this.userDao.save(user);
83          } catch (final ConstraintViolationException e) {
84              this.context.setRollbackOnly();
85              throw e;
86          }
87  
88          return savedUser;
89      }

91      /**
92       * Find a User by their Username. NoResultException thrown if the user is not found.
93       * 
94       * @param username
95       *            Username to search for
96       * @return User if found, null if not
97       */
98      public UserAccount findByUserEmail(final String username) {
99          final EntityManager entityManager = this.userDao.getEntityManager();
100         final CriteriaBuilder criteriaBuilder = entityManager.getCriteriaBuilder();
101 
102         final CriteriaQuery<UserAccount> criteriaQuery = criteriaBuilder.createQuery(UserAccount.class);
103         final Root<UserAccount> root = criteriaQuery.from(UserAccount.class);
104         criteriaQuery.select(root);
105 
106         final ParameterExpression<String> usernameParam = criteriaBuilder.parameter(String.class);
107         criteriaQuery.where(criteriaBuilder.equal(root.get("email"), usernameParam));
108 
109         final TypedQuery<UserAccount> query = entityManager.createQuery(criteriaQuery);
110         query.setParameter(usernameParam, username);
111 
112         UserAccount foundAccount = null;
113         try {
114             foundAccount = query.getSingleResult();
115         } catch (final NoResultException e) {
116             this.logger.info("Could not Find UserAccount for Username: " + username);
117         }
118 
119         return foundAccount;
120     }
