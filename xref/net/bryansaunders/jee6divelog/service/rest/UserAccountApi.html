69      /**
70       * Registers a User.
71       * 
72       * <ul>
73       * <li>Status 200: Successful Registration.</li>
74       * <li>Status 400: Error with the Request. Most likely caused by an invalid UserAccount.</li>
75       * </ul>
76       * 
77       * @param user
78       *            User to Register
79       * @return Registered User
80       */
81      @POST
82      @Path("register")
83      @TypeHint(UserAccount.class)
84      public Response registerUser(final UserAccount user) {
85          Response response;
86  
87          try {
88              UserAccount savedUser = this.userAccountService.createUser(user);
89              savedUser = SecurityUtils.getCleanUserAccount(savedUser);
90              response = Response.ok(savedUser).status(Response.Status.CREATED).build();
91          } catch (final EJBException e) {
92              response = Response.status(Response.Status.BAD_REQUEST).entity("JSON Invalid: " + e.getMessage()).build();
93          }
94  
95          return response;
96      }

104      * </ul>
105      * 
106      * @param example
107      *            Criteria to search for
108      * @return Found users
109      */
110     @POST
111     @Path("find")
112     @TypeHint(List.class)
113     @HasRole(role = Role.USER)
114     public Response findUser(final UserAccount example) {
115         Response response = null;
116 
117         List<UserAccount> results = this.userAccountService.findByExample(example);
118         if (results != null && !results.isEmpty()) {
119             response = Response.ok(results).build();
120         } else {
121             response = Response.status(Response.Status.BAD_REQUEST).entity("No Results Found.").build();
122         }
123 
124         return response;
125     }
