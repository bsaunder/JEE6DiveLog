47  
48          if (user == null) {
49              response = Response.status(Response.Status.BAD_REQUEST).entity("User Object must not be null").build();
50          } else {
51              UserAccount savedUser = this.userAccountService.createUser(user);
52              savedUser.setPassword("***"); // Clear out the Password
53              response = Response.ok(savedUser).status(Response.Status.CREATED).build();
54          }
55

57      }

70          Response response;
71  
72          final UserAccount foundUser = this.userAccountService.findByUserEmail(userName);
73          if (foundUser != null) {
