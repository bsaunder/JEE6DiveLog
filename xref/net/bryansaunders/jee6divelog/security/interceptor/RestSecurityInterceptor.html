51  /**
52   * Security Intercepter for REST API Calls.
53   * 
54   * @author Bryan Saunders <btsaunde@gmail.com>
55   * 
56   */
57  @ServerInterceptor
58  @Precedence("SECURITY")
59  public class RestSecurityInterceptor implements PreProcessInterceptor {
60  
61      /**
62       * Logger.
63       */
64      private final Logger logger = LoggerFactory.getLogger(RestSecurityInterceptor.class);
65  
66      /**
67       * Identity for current Session.
68       */
69      @Inject
70      private Identity identity;
71  
72      /**
73       * User Account Service.
74       */
75      @Inject
76      private UserAccountService userAccountService;
77  
78      /**
79       * Called before any REST service calls are processed.
80       * 
81       * @param httpRequest
82       *            Initial HTTP Request
83       * @param resourceMethod
84       *            Resource Method being Called
85       * @return Response for the Server, Null if the API call should proceed
86       * @throws Failure
87       *             Thrown on Failure
88       * @throws WebApplicationException
89       *             Thrown on Error
90       */
91      @Override
92      public ServerResponse preProcess(final HttpRequest httpRequest, final ResourceMethod resourceMethod)
93              throws Failure, WebApplicationException {
94          ServerResponse response = null;
95          String userName = null;
96          String userToken = null;
97  
98          this.logger.info("REST API Called: " + httpRequest.getUri().getAbsolutePath());
99  
100         final boolean isSecured = SecurityUtils.isMethodSecure(resourceMethod);
101 
102         if (isSecured) {
103             this.logger.debug("Secure API Call: " + httpRequest.getUri().getAbsolutePath());
104 
105             userName = this.getUserDefinedHeader(httpRequest, "dl-username");
106             userToken = this.getUserDefinedHeader(httpRequest, "dl-token");
107 
108             this.logger.debug("Using Headers: userName:" + userName + " | token:" + userToken);
109 
110             // Get UserAccount by Username
111             if (userName != null && userToken != null) {
112                 this.logger.debug("REST API Call Headers Valid");
113                 final UserAccount userAccount = this.userAccountService.findByUserEmail(userName);
114 
115                 if (userAccount != null) {
116                     this.logger.debug("REST API User Found");
117                     // Check Expiration Date
118                     final Date expirationDate = userAccount.getApiKeyExpiration();
119 
120                     if (expirationDate != null && System.currentTimeMillis() < expirationDate.getTime()) {
121                         this.logger.debug("API Token Not Expired");
122                         // Generate Token
123                         final String apiKey = userAccount.getApiKey();
124                         final String expectedToken = SecurityUtils.generateRestApiToken(userName, apiKey);
125 
126                         // Check Tokens
127                         if (expectedToken.equals(userToken)) {
128                             this.logger.debug("API Token Valid");
129 
130                             // Set Identity
131                             this.identity.setApiKey(apiKey);
132                             this.identity.setApiKeyExpiration(expirationDate);
133                             this.identity.setPermissions(userAccount.getPermissions());
134                             this.identity.setRoles(userAccount.getRoles());
135                             this.identity.setStatus(Identity.LOGGED_IN);
136                             final Credentials credentials = new Credentials();
137                             credentials.setUsername(userName);
138                             credentials.setPassword(userAccount.getPassword());
139                             this.identity.setCredentials(credentials);
140                             
141                             // Return null
142                             response = null;
143 
144                         } else {
145                             response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid Token");
146                         }
147                     } else {
148                         response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Token Expired");
149                     }
150                 } else {
151                     response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid User");
152                 }
153             } else {
154                 response = this.buildResponse(HttpResponseCodes.SC_BAD_REQUEST, "Missing Headers");
155             }
156         }
157 
158         return response;
159     }
160 
161     /**
162      * Build a ServerResponse Message.
163      * 
164      * @param statusCode
165      *            Response Status
166      * @param message
167      *            Response Message
168      * @return ServerResponse
169      */
170     private ServerResponse buildResponse(final int statusCode, final String message) {
171         this.logger.info("Creating REST Response: " + statusCode + " - " + message);
172         final ServerResponse response = new ServerResponse();
173         response.setStatus(statusCode);
174         response.setEntity(message);
175         return response;
176     }
177 
178     /**
179      * Gets a User Defined Header.
180      * 
181      * @param httpRequest
182      *            HTTP Request with Headers
183      * @param headerName
184      *            Name of User Defined Header
185      * @return Header Value if Found, null if not found
186      */
187     private String getUserDefinedHeader(final HttpRequest httpRequest, final String headerName) {
188         String headerValue = null;
189         final HttpHeaders headers = httpRequest.getHttpHeaders();
190 
191         final List<String> headerList = headers.getRequestHeader(headerName);
192         if (headerList != null && !headerList.isEmpty()) {
193             headerValue = headerList.get(0);
194         }
195 
196         return headerValue;
197     }
