54      /**
55       * Called before any REST service calls are processed.
56       * 
57       * @param httpRequest
58       *            Initial HTTP Request
59       * @param resourceMethod
60       *            Resource Method being Called
61       * @return Response for the Server, Null if the API call should proceed
62       * @throws Failure
63       *             Thrown on Failure
64       * @throws WebApplicationException
65       *             Thrown on Error
66       */
67      @Override
68      public ServerResponse preProcess(final HttpRequest httpRequest, final ResourceMethod resourceMethod)
69              throws Failure, WebApplicationException {
70          ServerResponse response = null;
71          String userName = null;
72          String userToken = null;
73  
74          this.logger.info("REST API Called: " + httpRequest.getUri().getAbsolutePath());
75  
76          final boolean isSecured = SecurityUtils.isMethodSecure(resourceMethod);
77  
78          if (isSecured) {
79              this.logger.debug("Secure API Call: " + httpRequest.getUri().getAbsolutePath());
80  
81              userName = this.getUserDefinedHeader(httpRequest, "dl-username");
82              userToken = this.getUserDefinedHeader(httpRequest, "dl-token");
83  
84              this.logger.debug("Using Headers: userName:" + userName + " | token:" + userToken);
85  
86              // Get UserAccount by Username
87              if (userName != null && userToken != null) {
88                  this.logger.debug("REST API Call Headers Valid");
89                  final UserAccount userAccount = this.userAccountService.findByUserEmail(userName);
90  
91                  if (userAccount != null) {
92                      this.logger.debug("REST API User Found");
93                      // Check Expiration Date
94                      final Date expirationDate = userAccount.getApiKeyExpiration();
95  
96                      if (expirationDate != null && System.currentTimeMillis() < expirationDate.getTime()) {
97                          this.logger.debug("API Token Not Expired");
98                          // Generate Token
99                          final String apiKey = userAccount.getApiKey();
100                         final String expectedToken = SecurityUtils.generateRestApiToken(userName, apiKey);
101 
102                         // Check Tokens
103                         if (expectedToken.equals(userToken)) {
104                             this.logger.debug("API Token Valid");
105 
106                             // Set Identity
107                             this.identity.setApiKey(apiKey);
108                             this.identity.setApiKeyExpiration(expirationDate);
109                             this.identity.setPermissions(userAccount.getPermissions());
110                             this.identity.setRoles(userAccount.getRoles());
111                             this.identity.setStatus(Identity.LOGGED_IN);
112                             final Credentials credentials = new Credentials();
113                             credentials.setUsername(userName);
114                             credentials.setPassword(userAccount.getPassword());
115                             this.identity.setCredentials(credentials);
116                             
117                             // Return null
118                             response = null;
119 
120                         } else {
121                             response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid Token");
122                         }
123                     } else {
124                         response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Token Expired");
125                     }
126                 } else {
127                     response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid User");
128                 }
129             } else {
130                 response = this.buildResponse(HttpResponseCodes.SC_BAD_REQUEST, "Missing Headers");
131             }
132         }
133 
134         return response;
135     }

137     /**

140      * @param statusCode
141      *            Response Status
142      * @param message
143      *            Response Message
144      * @return ServerResponse
145      */
146     private ServerResponse buildResponse(final int statusCode, final String message) {
147         this.logger.info("Creating REST Response: " + statusCode + " - " + message);
148         final ServerResponse response = new ServerResponse();
149         response.setStatus(statusCode);
150         response.setEntity(message);
151         return response;
152     }

154     /**
155      * Gets a User Defined Header.
156      * 
157      * @param httpRequest
158      *            HTTP Request with Headers
159      * @param headerName
160      *            Name of User Defined Header
161      * @return Header Value if Found, null if not found
162      */
163     private String getUserDefinedHeader(final HttpRequest httpRequest, final String headerName) {
164         String headerValue = null;
165         final HttpHeaders headers = httpRequest.getHttpHeaders();
166 
167         final List<String> headerList = headers.getRequestHeader(headerName);
168         if (headerList != null && !headerList.isEmpty()) {
169             headerValue = headerList.get(0);
170         }
171 
172         return headerValue;
173     }
