50      /**
51       * Called before any REST service calls are processed.
52       * 
53       * @param httpRequest
54       *            Initial HTTP Request
55       * @param resourceMethod
56       *            Resource Method being Called
57       * @return Response for the Server, Null if the API call should proceed
58       * @throws Failure
59       *             Thrown on Failure
60       * @throws WebApplicationException
61       *             Thrown on Error
62       */
63      @Override
64      public ServerResponse preProcess(final HttpRequest httpRequest, final ResourceMethod resourceMethod)
65              throws Failure, WebApplicationException {
66          ServerResponse response = null;
67          String userName = null;
68          String userToken = null;
69  
70          this.logger.info("REST API Called: " + httpRequest.getUri().getAbsolutePath());
71  
72          final boolean isSecured = SecurityUtils.isMethodSecure(resourceMethod);
73  
74          if (isSecured) {
75              this.logger.debug("Secure API Call: " + httpRequest.getUri().getAbsolutePath());
76  
77              userName = this.getUserDefinedHeader(httpRequest, "dl-username");
78              userToken = this.getUserDefinedHeader(httpRequest, "dl-token");
79  
80              this.logger.debug("Using Headers: userName:" + userName + " | token:" + userToken);
81  
82              // Get UserAccount by Username
83              if (userName != null && userToken != null) {
84                  this.logger.debug("REST API Call Headers Valid");
85                  final UserAccount userAccount = this.userAccountService.findByUserEmail(userName);
86  
87                  if (userAccount != null) {
88                      this.logger.debug("REST API User Found");
89                      // Check Expiration Date
90                      final Date expirationDate = userAccount.getApiKeyExpiration();
91  
92                      if (expirationDate != null && System.currentTimeMillis() < expirationDate.getTime()) {
93                          this.logger.debug("API Token Not Expired");
94                          // Generate Token
95                          final String apiKey = userAccount.getApiKey();
96                          final String expectedToken = SecurityUtils.generateRestApiToken(userName, apiKey);
97  
98                          // Check Tokens
99                          if (expectedToken.equals(userToken)) {
100                             this.logger.debug("API Token Valid");
101 
102                             // Set Identity
103                             this.identity.setApiKey(apiKey);
104                             this.identity.setApiKeyExpiration(expirationDate);
105                             this.identity.setPermissions(userAccount.getPermissions());
106                             this.identity.setRoles(userAccount.getRoles());
107                             this.identity.setStatus(Identity.LOGGED_IN);
108                             final Credentials credentials = new Credentials();
109                             credentials.setUsername(userName);
110                             credentials.setPassword(userAccount.getPassword());
111                             this.identity.setCredentials(credentials);
112                             
113                             // Return null
114                             response = null;
115 
116                         } else {
117                             response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid Token");
118                         }
119                     } else {
120                         response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Token Expired");
121                     }
122                 } else {
123                     response = this.buildResponse(HttpResponseCodes.SC_UNAUTHORIZED, "Invalid User");
124                 }
125             } else {
126                 response = this.buildResponse(HttpResponseCodes.SC_BAD_REQUEST, "Missing Headers");
127             }
128         }
129 
130         return response;
131     }

133     /**

135      * 
136      * @param statusCode
137      *            Response Status
138      * @param message
139      *            Response Message
140      * @return ServerResponse
141      */
142     private ServerResponse buildResponse(final int statusCode, final String message) {
143         this.logger.info("Creating REST Response: " + statusCode + " - " + message);
144         final ServerResponse response = new ServerResponse();
145         response.setStatus(statusCode);
146         response.setEntity(message);
147         return response;
148     }

150     /**
151      * Gets a User Defined Header.
152      * 
153      * @param httpRequest
154      *            HTTP Request with Headers
155      * @param headerName
156      *            Name of User Defined Header
157      * @return Header Value if Found, null if not found
158      */
159     private String getUserDefinedHeader(final HttpRequest httpRequest, final String headerName) {
160         String headerValue = null;
161         final HttpHeaders headers = httpRequest.getHttpHeaders();
162 
163         final List<String> headerList = headers.getRequestHeader(headerName);
164         if (headerList != null && !headerList.isEmpty()) {
165             headerValue = headerList.get(0);
166         }
167 
168         return headerValue;
169     }
