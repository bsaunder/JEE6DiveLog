31      /**
32       * Intercepts the Method Call and Throws an Exception if Access is Not Allowed.
33       * 
34       * @param ctx
35       *            Invocation Context
36       * @return Result
37       * @throws Exception
38       *             Thrown if Access is Denied
39       */
40      @AroundInvoke
41      public Object checkAuthenticaion(final InvocationContext ctx) throws Exception {
42          final Method method = ctx.getMethod();
43          Role role = null;
44  
45          // Check For Method Level Annotations
46          HasRole hasRoleAnnotation = method.getAnnotation(HasRole.class);
47          if (hasRoleAnnotation != null) {
48              role = hasRoleAnnotation.role();
49          }
50  
51          // Check For Class Level Annotations
52          final Class<?> callingClass = method.getDeclaringClass();
53          if (callingClass != null) {
54              hasRoleAnnotation = callingClass.getAnnotation(HasRole.class);
55              if (hasRoleAnnotation != null) {
56                  role = hasRoleAnnotation.role();
57              }
58          }
59  
60          // Check Identity
61          if (role == null) {
62              throw new AuthorizationFailedException("Could not Validate Identity, Could not Retrieve Role.");
63          } else {
64              if (!this.identity.hasRole(role)) {
65                  throw new AuthorizationFailedException("Access Denied, Role " + role + " Not Found");
66              }
67          }
68  
69          return ctx.proceed();
70      }
