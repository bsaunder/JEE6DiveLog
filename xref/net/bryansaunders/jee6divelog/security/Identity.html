114     /**
115      * Login Identity.
116      * 
117      * @return Login Result
118      */
119     public boolean login() {
120         this.logger.info("Logging In User with Credentials: " + this.credentials);
121         boolean returnValue = Identity.LOGIN_FAILURE;
122 
123         final String username = this.getUsername();
124         final String password = this.getPassword();
125         final String expectedPassword = SecurityUtils.generatePasswordHash(password);
126 
127         final UserAccount userAccount = this.userAccountService.findByUserEmail(username);
128         if (userAccount != null) {
129             final String hashedPassword = userAccount.getPassword();
130             if (hashedPassword.equals(expectedPassword)) {
131                 this.setStatus(Identity.LOGGED_IN);
132 
133                 // Set Roles
134                 // Set Permissions
135 
136                 this.logger.info("Logged in User: " + this.credentials);
137             }
138         }
139 
140         if (this.isLoggedIn()) {
141             returnValue = Identity.LOGIN_SUCCESS;
142         } else {
143             this.logger.warn("Invalid Credentials: " + this.credentials);
144         }
145 
146         return returnValue;
147     }
