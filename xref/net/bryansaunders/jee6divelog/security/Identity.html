178     /**
179      * Login Identity.
180      * 
181      * @param restLogin
182      *            Is it a REST Login
183      * @return Login Result
184      */
185     public boolean login(final boolean restLogin) {
186         this.logger.debug("Logging In User with Credentials: " + this.credentials);
187         boolean returnValue = Identity.LOGIN_FAILURE;
188 
189         final String username = this.getUsername();
190         final String password = this.getPassword();
191         final String expectedPassword = SecurityUtils.generatePasswordHash(password);
192 
193         final UserAccount userAccount = this.userAccountService.findByUserEmail(username);
194         if (userAccount != null) {
195             final String hashedPassword = userAccount.getPassword();
196             if (hashedPassword.equals(expectedPassword)) {
197                 this.setStatus(Identity.LOGGED_IN);
198 
199                 this.setRoles(userAccount.getRoles());
200                 this.setPermissions(userAccount.getPermissions());
201 
202                 if (restLogin) {
203                     this.publicApiKey = SecurityUtils.generateRestApiKey();
204                     this.privateApiKey = SecurityUtils.generateRestApiKey();
205 
206                     final Calendar expiration = Calendar.getInstance();
207                     expiration.add(Calendar.HOUR, 5);
208                     this.apiKeyExpiration = expiration.getTime();
209 
210                     userAccount.setPublicApiKey(this.publicApiKey);
211                     userAccount.setPrivateApiKey(this.privateApiKey);
212                     userAccount.setApiKeyExpiration(this.apiKeyExpiration);
213                     this.userAccountService.saveUser(userAccount);
214                 }
215 
216                 this.logger.debug("Logged in User: " + this.credentials);
217             }
218         }
219 
220         if (this.isLoggedIn()) {
221             returnValue = Identity.LOGIN_SUCCESS;
222         } else {
223             this.logger.warn("Invalid Credentials: " + this.credentials);
224         }
225 
226         return returnValue;
227     }

229     /**

232      * @return Logout Result

234     public String logout() {
235         this.logger.info("Logging Out User with Credentials: " + this.credentials);
236         String returnValue = Identity.LOGOUT_FAILURE;
237 
238         this.setStatus(Identity.LOGGED_OUT);
239 
240         if (!this.isLoggedIn()) {
241             returnValue = Identity.LOGOUT_SUCCESS;
242         }
243 
244         return returnValue;
245     }
