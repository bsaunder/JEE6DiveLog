73      /**
74       * Tests Valid Login.
75       */
76      @Test
77      public void ifLoginValidThenSuccess() {
78          // given
79          final String email = "joe@test.com";
80          final String pass = "joe123";
81          final boolean rememberMe = true;
82          
83          this.loginBean.setEmail(email);
84          this.loginBean.setPassword(pass);
85          this.loginBean.setRememberMe(rememberMe);
86          
87          final Identity identity = mock(Identity.class);
88          final Credentials credentials = new Credentials();
89          this.loginBean.setIdentity(identity);
90          this.loginBean.setCredentials(credentials);
91          
92          // when
93          doCallRealMethod().when(identity).setRememberMe(anyBoolean());
94          doCallRealMethod().when(identity).isRememberMe();
95          doReturn(Identity.LOGIN_SUCCESS).when(identity).login();
96          final String result = this.loginBean.login();
97          
98          // then
99          assertEquals(LoginBean.SUCCESS, result);
100         assertEquals(email, credentials.getUsername());
101         assertEquals(pass, credentials.getPassword());
102         assertEquals(rememberMe, identity.isRememberMe());
103     }
