146     @SuppressWarnings("unchecked")
147     @Test
148     public void ifValidThenAuthorized() {
149         // given
150         final String username = this.userAccount.getEmail();
151         final String apiKey = this.userAccount.getApiKey();
152         final String apiToken = SecurityUtils.generateRestApiToken(username, apiKey);
153 
154         final MultivaluedMap<String, String> requestHeaders = new MultivaluedMapImpl<String, String>();
155         requestHeaders.add("dl-username", username);
156         requestHeaders.add("dl-token", apiToken);
157         this.httpHeaders.setRequestHeaders(requestHeaders);
158 
159         final ResourceMethod resourceMethod = mock(ResourceMethod.class);
160         when(resourceMethod.getMethod()).thenReturn(this.secureMethod);
161 
162         when(this.mockAccountService.findByUserEmail(any(String.class))).thenReturn(this.userAccount);
163 
164         // when
165         final ServerResponse response = this.interceptor.preProcess(this.httpRequest, resourceMethod);
166 
167         // then
168         assertNull(response);
169         verify(this.mockIdentity).setApiKey(any(String.class));
170         verify(this.mockIdentity).setApiKeyExpiration(any(Date.class));
171         verify(this.mockIdentity).setPermissions(any(Set.class));
172         verify(this.mockIdentity).setRoles(any(Set.class));
173         verify(this.mockIdentity).setStatus(anyInt());
174         verify(this.mockIdentity).setCredentials(any(Credentials.class));
