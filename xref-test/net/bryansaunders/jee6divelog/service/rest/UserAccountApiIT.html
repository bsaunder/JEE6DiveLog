109     /**
110      * Test find by username with Valid Username.
111      * 
112      * @throws Exception
113      *             Thrown on Error
114      */
115     @Test
116     public void ifSingleCriteriaMatchesThenGetResults() throws Exception {
117         final String token = this.doLogin(UserAccountApiIT.VALID_EMAIL, UserAccountApiIT.VALID_PASSWORD);
118 
119         final UserAccount userAccount = new UserAccount();
120         userAccount.setEmail(UserAccountApiIT.VALID_EMAIL);
121 
122         final String json = given()
123                 .headers(UserAccountApiIT.DL_USERNAME, UserAccountApiIT.VALID_EMAIL, UserAccountApiIT.DL_TOKEN, token)
124                 .contentType(ContentType.JSON).body(userAccount).expect().statusCode(RestApiTest.OK).when()
125                 .post(RestApiTest.URL_ROOT + "/user/find/").asString();
126 
127         assertNotNull(json);
128 
129         final ObjectMapper objMapper = new ObjectMapper();
130         final List<UserAccount> list = objMapper.readValue(json, new TypeReference<List<UserAccount>>() {
131         });
132         assertNotNull(list);
133         assertEquals(1, list.size());
134 
135         final UserAccount foundUser = list.get(0);
136         assertNotNull(foundUser);
137         assertEquals(UserAccountApiIT.VALID_EMAIL, foundUser.getEmail());
138     }
