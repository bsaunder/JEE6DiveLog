86       * Performs a Login Action and returns the Authorization Token.
87       * 
88       * @param userName
89       *            User Name
90       * @param password
91       *            User Password
92       * @return Authorization Token
93       */
94      protected UserAccount doLogin(final String userName, final String password) {
95          final Credentials credentials = new Credentials();
96          credentials.setUsername(userName);
97          credentials.setPassword(password);
98  
99          final UserAccount user = given().contentType(ContentType.JSON).body(credentials).expect()
100                 .statusCode(HttpStatus.SC_ACCEPTED).when().post(RestApiTest.URL_ROOT + "/security/login/")
101                 .as(UserAccount.class);
102 
103         assertNotNull(user);
104         return user;
105     }

107     /**
108      * Generates Headers for the Logged In User.
109      * 
110      * @param requestMethod
111      *            HTTP Request Method
112      * @param requestUrl
113      *            Request URL
114      * @param privateApiKey
115      *            Private API Key
116      * @param publicApiKey
117      *            Public API Key
118      * @param jsonContent
119      *            JSON Content String
120      * @return Header Map
121      */
122     protected Map<String, String> generateLoginHeaders(final String requestMethod, final String requestUrl,
123             final String jsonContent, final String privateApiKey, final String publicApiKey) {
124         final Map<String, String> headerMap = new HashMap<String, String>();
125 
126         final String date = new SimpleDateFormat("yyyy-MM-dd").format(new Date());
127         final String contentType = MediaType.APPLICATION_JSON;
128         final String contentMd5 = "";
129         if (jsonContent != null) {
130             DigestUtils.md5(jsonContent);
131         }
132 
133         final String signature = SecurityUtils.generateRestSignature(requestMethod, contentType, contentMd5, date,
134                 requestUrl, privateApiKey);
135 
136         headerMap.put(HttpHeaders.DATE, date);
137         headerMap.put(HttpHeaders.CONTENT_TYPE, contentType);
138         headerMap.put(RestApi.CONTENT_MD5_HEADER, contentMd5);
139         headerMap.put(RestApi.PUBLIC_KEY_HEADER, publicApiKey);
140         headerMap.put(RestApi.SIGNATURE_HEADER, signature);
141 
142         return headerMap;
143     }
